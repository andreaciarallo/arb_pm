```python
# Field name on CLOB market objects that groups markets by event.
# Markets in the same event are mutually exclusive — exactly one resolves YES.
_EVENT_FIELD = "<confirmed_field_name>"          # from Step 1 discovery
_EVENT_FIELD_ALT = "neg_risk_market_id"          # fallback for NegRisk-only markets


def _group_by_event(markets: list[dict]) -> list[list[dict]]:
    """
    Group markets by Polymarket event ID.

    All markets in the same event are mutually exclusive — exactly one
    will resolve YES. This covers both NegRisk-enabled and standard
    multi-outcome events (e.g., "Who wins the election?" with N candidates).

    Markets without an event ID are standalone binary markets and ignored.
    """
    groups: dict[str, list[dict]] = defaultdict(list)
    for market in markets:
        event_id = market.get(_EVENT_FIELD) or market.get(_EVENT_FIELD_ALT)
        if event_id:
            groups[event_id].append(market)
    return [
        group for group in groups.values()
        if _MIN_GROUP_SIZE <= len(group) <= _MAX_GROUP_SIZE
    ]
```

```python
import httpx

_GAMMA_EVENTS_URL = "https://gamma-api.polymarket.com/events"
_event_groups: dict[str, str] = {}  # condition_id → event_id, populated at startup


def load_event_groups(condition_ids: list[str]) -> None:
    """
    Fetch event→market mappings from gamma API and populate _event_groups.
    Call this once when markets are loaded, not on every detection cycle.
    condition_ids: list of CLOB condition_ids to look up.
    """
    global _event_groups
    try:
        resp = httpx.get(
            _GAMMA_EVENTS_URL,
            params={"active": "true", "limit": 500},
            timeout=10.0,
        )
        resp.raise_for_status()
        for event in resp.json():
            event_id = str(event.get("id", ""))
            for market in event.get("markets", []):
                cid = market.get("conditionId") or market.get("condition_id", "")
                if cid:
                    _event_groups[cid] = event_id
    except Exception as exc:
        logger.warning(f"load_event_groups: gamma API fetch failed: {exc}")


def _group_by_event(markets: list[dict]) -> list[list[dict]]:
    """
    Group markets by Polymarket event (from gamma API) or neg_risk_market_id fallback.
    Call load_event_groups() once before using this function.
    """
    groups: dict[str, list[dict]] = defaultdict(list)
    for market in markets:
        cid = market.get("condition_id", "")
        event_id = (
            _event_groups.get(cid)                    # gamma API event
            or market.get("neg_risk_market_id")        # NegRisk fallback
            or market.get("neg_risk_id")               # alt field name
        )
        if event_id:
            groups[event_id].append(market)
    return [
        group for group in groups.values()
        if _MIN_GROUP_SIZE <= len(group) <= _MAX_GROUP_SIZE
    ]
```
