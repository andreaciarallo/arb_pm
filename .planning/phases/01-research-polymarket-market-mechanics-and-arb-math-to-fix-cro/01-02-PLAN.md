```python
# Field name Polymarket uses for neg-risk group membership.
# Markets sharing this ID are guaranteed mutually exclusive by Polymarket.
_NEG_RISK_FIELD = "neg_risk_market_id"
_NEG_RISK_FIELD_ALT = "neg_risk_id"   # fallback in case of API naming variation


def _group_by_neg_risk(markets: list[dict]) -> list[list[dict]]:
    """
    Group markets by Polymarket's official neg_risk_market_id.

    Markets sharing the same neg_risk_market_id are guaranteed mutually
    exclusive by Polymarket — no keyword heuristic or LLM needed.

    Markets without a neg_risk_market_id are ignored (they are standalone
    binary markets, not part of an N-way exclusive set).
    """
    groups: dict[str, list[dict]] = defaultdict(list)
    for market in markets:
        neg_id = market.get(_NEG_RISK_FIELD) or market.get(_NEG_RISK_FIELD_ALT)
        if neg_id:
            groups[neg_id].append(market)
    return [
        group for group in groups.values()
        if _MIN_GROUP_SIZE <= len(group) <= _MAX_GROUP_SIZE
    ]
```

```python
def test_neg_risk_grouping():
    """Markets sharing neg_risk_market_id are grouped and produce a cross_market opportunity."""
    from bot.scanner.price_cache import PriceCache
    from bot.detection.cross_market import detect_cross_market_opportunities

    cache = PriceCache()
    markets = [
        _make_market("0x1", "Will Alice win?", "tok_a", neg_risk_market_id="nrm_001"),
        _make_market("0x2", "Will Bob win?", "tok_b", neg_risk_market_id="nrm_001"),
```
