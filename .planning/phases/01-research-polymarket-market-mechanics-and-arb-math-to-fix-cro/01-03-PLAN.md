Add this function BEFORE `execute_opportunity()` (so it can be called from within):

```python
async def _execute_cross_market(
    client,
    opp: "ArbitrageOpportunity",
    config,
    risk_gate,
    arb_id: str,
    results: list,
) -> tuple[str, list]:
    """
    Execute a cross-market arbitrage opportunity by placing one FAK BUY per leg.

    Each leg is {"token_id": str, "ask": float, "depth": float}.
    Size per leg is a flat fraction of available capital (equal weight across legs).
    Does NOT use the YES+NO Kelly sizing or retry-then-hedge path.

    Returns (arb_id, results) with one ExecutionResult per leg.
    """
    n_legs = len(opp.legs)
    # Equal capital allocation per leg; use kelly_max_capital_pct of total capital total,
    # divided evenly across legs so total exposure = kelly_max_capital_pct * capital.
    per_leg_usd = (config.total_capital_usd * config.kelly_max_capital_pct) / n_legs

    logger.info(
        f"Cross-market execution | market={opp.market_id} legs={n_legs} "
        f"per_leg_usd={per_leg_usd:.2f}"
    )

    for i, leg in enumerate(opp.legs):
        token_id = leg["token_id"]
        ask_price = leg["ask"]

        if risk_gate.is_kill_switch_active():
            logger.warning(
                f"Kill switch active — aborting cross-market legs at leg {i + 1}/{n_legs} "
                f"for market={opp.market_id}"
            )
            break

        leg_resp = None
        try:
            leg_resp = await place_fak_order(
                client, token_id, ask_price, per_leg_usd, "BUY"
            )
        except Exception as exc:
            logger.error(
                f"Cross-market leg {i + 1}/{n_legs} exception for market={opp.market_id}: {exc}"
            )

        results.append(ExecutionResult(
            market_id=opp.market_id,
            leg=f"leg_{i + 1}",
            side="BUY",
            token_id=token_id,
            price=ask_price,
            size=per_leg_usd,
            order_id=leg_resp.get("orderID") if leg_resp else None,
            status="filled" if leg_resp else "failed",
            size_filled=per_leg_usd if leg_resp else 0.0,
            kelly_size_usd=per_leg_usd,
            vwap_price=ask_price,
            error_msg=None if leg_resp else "place_fak_order returned None",
        ))

        if leg_resp:
            logger.info(
                f"Cross-market leg {i + 1}/{n_legs} filled | "
                f"market={opp.market_id} token={token_id} order={leg_resp.get('orderID')}"
            )
        else:
            logger.warning(
                f"Cross-market leg {i + 1}/{n_legs} failed | market={opp.market_id}"
            )

    return arb_id, results
```
